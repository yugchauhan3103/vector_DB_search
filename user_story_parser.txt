"""
Text Extraction Service
Supports .docx (python-docx) and .pdf (pdfplumber / PyMuPDF fallback).

"""

import io
import logging
from pathlib import Path

logger = logging.getLogger(__name__)

SUPPORTED_EXTENSIONS = {".docx", ".pdf"}
SUPPORTED_MIME_TYPES = {
    "application/vnd.openxmlformats-officedocument.wordprocessingml.document",
    "application/pdf",
    "application/octet-stream",  # browsers sometimes send generic MIME
}


def extract_text_from_file(file_bytes: bytes, filename: str, content_type: str) -> str:
    """
    Detect file type and extract plain text.

    Args:
        file_bytes: Raw bytes of the uploaded file.
        filename:   Original filename (used to detect extension).
        content_type: MIME type reported by the browser.

    Returns:
        Extracted plain text as a string.

    Raises:
        ValueError: If file type is not supported.
    """
    ext = Path(filename).suffix.lower()

    if ext == ".docx" or "wordprocessingml" in (content_type or ""):
        logger.info("Extracting text from DOCX: %s", filename)
        return _extract_from_docx(file_bytes)

    if ext == ".pdf" or content_type == "application/pdf":
        logger.info("Extracting text from PDF: %s", filename)
        return _extract_from_pdf(file_bytes)

    raise ValueError(
        f"Unsupported file type '{ext}'. Only .docx and .pdf are supported."
    )


# ---------------------------------------------------------------------------
# DOCX extraction
# ---------------------------------------------------------------------------

def _extract_from_docx(file_bytes: bytes) -> str:
    """Extract text from a .docx file using python-docx."""
    try:
        from docx import Document  # python-docx
    except ImportError:
        raise RuntimeError(
            "python-docx is not installed. Run: pip install python-docx"
        )

    doc = Document(io.BytesIO(file_bytes))
    parts: list[str] = []

    for para in doc.paragraphs:
        text = para.text.strip()
        if text:
            parts.append(text)

    # Also extract text from tables (important for user stories with tabular data)
    for table in doc.tables:
        for row in table.rows:
            row_cells = [cell.text.strip() for cell in row.cells if cell.text.strip()]
            if row_cells:
                parts.append(" | ".join(row_cells))

    return "\n".join(parts)

# PDF extraction
# ---------------------------------------------------------------------------

def _extract_from_pdf(file_bytes: bytes) -> str:
    """
    Extract text from a PDF.
    Tries pdfplumber first (better layout), falls back to PyMuPDF (fitz).
    """
    text = _extract_pdf_pdfplumber(file_bytes)
    if text and text.strip():
        return text

    logger.warning("pdfplumber returned empty text; trying PyMuPDF fallback.")
    text = _extract_pdf_pymupdf(file_bytes)
    if text and text.strip():
        return text

    raise ValueError(
        "Could not extract any text from the PDF. "
        "The file may be scanned/image-based. Please use a text-selectable PDF."
    )


def _extract_pdf_pdfplumber(file_bytes: bytes) -> str:
    try:
        import pdfplumber
    except ImportError:
        logger.warning("pdfplumber not installed; skipping. Run: pip install pdfplumber")
        return ""

    pages: list[str] = []
    with pdfplumber.open(io.BytesIO(file_bytes)) as pdf:
        for page in pdf.pages:
            page_text = page.extract_text()
            if page_text:
                pages.append(page_text.strip())
    return "\n\n".join(pages)


def _extract_pdf_pymupdf(file_bytes: bytes) -> str:
    try:
        import fitz  # PyMuPDF
    except ImportError:
        logger.warning("PyMuPDF not installed; skipping. Run: pip install PyMuPDF")
        return ""

    doc = fitz.open(stream=file_bytes, filetype="pdf")
    pages: list[str] = []
    for page in doc:
        pages.append(page.get_text())
    doc.close()
    return "\n\n".join(pages)

LLM_API_KEY = os.getenv("OPENAI_API_KEY")

MODEL_PRICING: dict[str, dict[str, float]] = {
    "gpt-4o-mini": {"input": 0.000150, "output": 0.000600},   # per 1K tokens
    "gpt-4o":      {"input": 0.002500, "output": 0.010000},
    "gpt-4-turbo": {"input": 0.010000, "output": 0.030000},
    "gpt-3.5-turbo": {"input": 0.000500, "output": 0.001500},
}
 
 
def _calculate_cost(model: str, input_tokens: int, output_tokens: int) -> dict[str, Any]:
    """Calculate cost based on model pricing per 1K tokens."""
    pricing = MODEL_PRICING.get(model, MODEL_PRICING.get("gpt-4o-mini"))
    input_cost  = (input_tokens  / 1000) * pricing["input"]
    output_cost = (output_tokens / 1000) * pricing["output"]
    total_cost  = input_cost + output_cost
 
    return {
        "input_tokens":  input_tokens,
        "output_tokens": output_tokens,
        "total_tokens":  input_tokens + output_tokens,
        "input_cost_usd":  round(input_cost,  6),
        "output_cost_usd": round(output_cost, 6),
        "total_cost_usd":  round(total_cost,  6),
        "model": model,
    }

# Public API
# ---------------------------------------------------------------------------

async def extract_requirements_from_text(user_story_text: str) -> dict[str, Any]:
    """
    Send user story text to the LLM and return parsed requirements as a dict.

    Raises:
        RuntimeError: if the LLM call fails or returns non-JSON.
    """
    prompt = USER_PROMPT_TEMPLATE.format(user_story_text=user_story_text)

    raw, usage = await _call_openai(prompt)

    requirements = _parse_json_response(raw)
 
    return {
        "requirements": requirements,
        "usage": usage,
    }


# ---------------------------------------------------------------------------
# Provider adapters
# ---------------------------------------------------------------------------

async def _call_openai(prompt: str) -> str:
    """Call OpenAI Chat Completions API."""
    url = url = "https://api.openai.com/v1/chat/completions"

    headers = {
        "Authorization": f"Bearer {LLM_API_KEY}",
        "Content-Type": "application/json",
    }

    payload = {
        "model": LLM_MODEL,
        "messages": [
            {"role": "system", "content": SYSTEM_PROMPT},
            {"role": "user",   "content": prompt},
        ],
        "temperature": 0.1,
        "max_tokens": 8192,
        "response_format": {"type": "json_object"},
    }

    async with httpx.AsyncClient(timeout=LLM_TIMEOUT) as client:
        resp = await client.post(url, json=payload, headers=headers)
        resp.raise_for_status()
        data = resp.json()

    content = data["choices"][0]["message"]["content"]
 
    # OpenAI always returns usage in the response
    raw_usage = data.get("usage", {})
    input_tok = raw_usage.get("prompt_tokens", 0)
    output_tok = raw_usage.get("completion_tokens", 0)
    usage = _calculate_cost(LLM_MODEL, input_tok, output_tok)
 
    return content, usage


# ---------------------------------------------------------------------------
# JSON parsing
# ---------------------------------------------------------------------------

def _parse_json_response(raw: str) -> dict[str, Any]:
    """
    Parse LLM response to a Python dict.
    Strips markdown code fences if present.
    """
    if not raw or not raw.strip():
        raise RuntimeError("LLM returned an empty response.")

    cleaned = re.sub(r"^```(?:json)?\s*", "", raw.strip(), flags=re.IGNORECASE)
    cleaned = re.sub(r"\s*```$", "", cleaned.strip())

    try:
        return json.loads(cleaned)
    except json.JSONDecodeError as e:
        logger.error("LLM response was not valid JSON:\n%s", raw[:500])
        raise RuntimeError(
            f"LLM did not return valid JSON. Parse error: {e}. "
            f"Raw response (first 300 chars): {raw[:300]}"
        )

"""
User Story Parser Router
Handles file upload (docx/pdf), text extraction, and LLM-based requirement extraction.
"""

import logging
from fastapi import APIRouter, File, UploadFile, HTTPException, status
from fastapi.responses import JSONResponse

from user_story_parser.extractor import extract_text_from_file
from user_story_parser.llm_service import extract_requirements_from_text
from user_story_parser.validators import validate_uploaded_file

logger = logging.getLogger(__name__)

router = APIRouter(prefix="/user-story", tags=["User Story Parser"])


@router.post(
    "/parse",
    summary="Parse a user story document and extract structured requirements",
    description=(
        "Upload a user story document (.docx or .pdf). "
        "The API extracts the text, passes it through an LLM with a structured prompt, "
        "and returns a detailed JSON describing screens, components, fields, validations, "
        "business rules, and more."
    ),
    response_description="Structured JSON representation of the user story requirements",
)

async def parse_user_story(
    file: UploadFile = File(..., description="User story document (.docx or .pdf)"),
):
    # Step 1: Validate
    validate_uploaded_file(file)
 
    # Step 2: Read file bytes
    try:
        file_bytes = await file.read()
    except Exception as e:
        logger.error("Failed to read uploaded file: %s", e)
        raise HTTPException(
            status_code=status.HTTP_400_BAD_REQUEST,
            detail=f"Could not read uploaded file: {str(e)}",
        )
 
    # Step 3: Extract text
    try:
        extracted_text = extract_text_from_file(
            file_bytes=file_bytes,
            filename=file.filename,
            content_type=file.content_type,
        )
    except ValueError as e:
        raise HTTPException(status_code=status.HTTP_422_UNPROCESSABLE_ENTITY, detail=str(e))
    except Exception as e:
        logger.error("Text extraction failed: %s", e)
        raise HTTPException(
            status_code=status.HTTP_500_INTERNAL_SERVER_ERROR,
            detail=f"Text extraction failed: {str(e)}",
        )
 
    if not extracted_text or not extracted_text.strip():
        raise HTTPException(
            status_code=status.HTTP_422_UNPROCESSABLE_ENTITY,
            detail="No text could be extracted from the uploaded document.",
        )
 
    # Step 4: LLM extraction
    try:
        result = await extract_requirements_from_text(extracted_text)
    except Exception as e:
        logger.error("LLM extraction failed: %s", e)
        raise HTTPException(
            status_code=status.HTTP_500_INTERNAL_SERVER_ERROR,
            detail=f"Requirement extraction via LLM failed: {str(e)}",
        )
 
    # Step 5: Return requirements + usage + cost
    return JSONResponse(
        content={
            "status": "success",
            "filename": file.filename,
            "requirements": result["requirements"],
            "usage": result["usage"],
        }
    )

validator.py
"""
File validation utilities for uploaded user story documents.
"""

from fastapi import HTTPException, UploadFile, status

# 20 MB limit
MAX_FILE_SIZE_BYTES = 20 * 1024 * 1024

ALLOWED_EXTENSIONS = {".docx", ".pdf"}

ALLOWED_CONTENT_TYPES = {
    "application/pdf",
    "application/vnd.openxmlformats-officedocument.wordprocessingml.document",
    "application/octet-stream",  # some browsers send this for .docx
}


def validate_uploaded_file(file: UploadFile) -> None:
    """
    Validate the uploaded file for type and basic sanity.

    Raises:
        HTTPException 400 if validation fails.
    """
    if not file.filename:
        raise HTTPException(
            status_code=status.HTTP_400_BAD_REQUEST,
            detail="No filename provided.",
        )

    from pathlib import Path
    ext = Path(file.filename).suffix.lower()

    if ext not in ALLOWED_EXTENSIONS:
        raise HTTPException(
            status_code=status.HTTP_400_BAD_REQUEST,
            detail=(
                f"Unsupported file type '{ext}'. "
                f"Allowed types: {', '.join(sorted(ALLOWED_EXTENSIONS))}"
            ),
        )

    # Content-type check (lenient — browsers can send generic types)
    content_type = (file.content_type or "").lower()
    if content_type and content_type not in ALLOWED_CONTENT_TYPES:
        # Warn but don't hard-block — some valid .docx come as octet-stream
        import logging
        logging.getLogger(__name__).warning(
            "Unexpected content_type '%s' for file '%s'. Proceeding by extension.",
            content_type, file.filename,
        )

Material Consumption (Against Job)
1. Purpose
The Material Consumption Module is used to record the actual quantity of material consumed by the Production Team after the material has already been issued against a specific Job.
This module helps the organization track:
•	total material issued to production 
•	actual material consumed in stages 
•	material returned back to warehouse 
•	remaining balance still lying with production 
This module is important because issued quantity and consumed quantity are not always the same.
________________________________________
2. Core Concept
The process flow is:
Material Issue With Job → Material goes to Production Team → Production consumes material in parts → Unused balance may remain with production or be returned to warehouse
So this module should only work on already issued material.
It should not create a fresh stock issue.

________________________________________
10.3 Remaining Quantity Behavior
Remaining quantity tells the user how much quantity is still with production and not yet consumed or returned.
Example
•	Issued = 10 
•	Consumed = 5 
•	Returned = 2 
Then:
Remaining = 10 – 5 – 2 = 3
That means 3 is still lying with production.

Formula 2 – Available Consumption Balance

Formula 3 – Remaining Quantity
Remaining Quantity = Issued Quantity – Consumed Quantity – Returned Quantity
