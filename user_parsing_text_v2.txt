"""
Text Extraction Service
Supports .docx (python-docx) and .pdf (pdfplumber / PyMuPDF fallback).

"""

import io
import logging
from pathlib import Path

logger = logging.getLogger(__name__)

SUPPORTED_EXTENSIONS = {".docx", ".pdf"}
SUPPORTED_MIME_TYPES = {
    "application/vnd.openxmlformats-officedocument.wordprocessingml.document",
    "application/pdf",
    "application/octet-stream",  # browsers sometimes send generic MIME
}


def extract_text_from_file(file_bytes: bytes, filename: str, content_type: str) -> str:
    """
    Detect file type and extract plain text.

    Args:
        file_bytes: Raw bytes of the uploaded file.
        filename:   Original filename (used to detect extension).
        content_type: MIME type reported by the browser.

    Returns:
        Extracted plain text as a string.

    Raises:
        ValueError: If file type is not supported.
    """
    ext = Path(filename).suffix.lower()

    if ext == ".docx" or "wordprocessingml" in (content_type or ""):
        logger.info("Extracting text from DOCX: %s", filename)
        return _extract_from_docx(file_bytes)

    if ext == ".pdf" or content_type == "application/pdf":
        logger.info("Extracting text from PDF: %s", filename)
        return _extract_from_pdf(file_bytes)

    raise ValueError(
        f"Unsupported file type '{ext}'. Only .docx and .pdf are supported."
    )


# ---------------------------------------------------------------------------
# DOCX extraction
# ---------------------------------------------------------------------------

def _extract_from_docx(file_bytes: bytes) -> str:
    """
    Extract text from a .docx file preserving document structure.

    Headings are emitted as Markdown-style markers (# / ## / ###).
    List items are prefixed with "- ".
    Tables are rendered in Markdown table format.
    Paragraphs and tables are interleaved in document order so the LLM
    can properly associate content with its surrounding context.
    """
    try:
        from docx import Document  # python-docx
    except ImportError:
        raise RuntimeError(
            "python-docx is not installed. Run: pip install python-docx"
        )

    doc = Document(io.BytesIO(file_bytes))

    # Build lookup maps keyed by XML element for document-order iteration
    para_map = {p._element: p for p in doc.paragraphs}
    table_map = {t._element: t for t in doc.tables}

    parts: list[str] = []

    for child in doc.element.body:
        if child in para_map:
            para = para_map[child]
            text = para.text.strip()
            if not text:
                continue

            style_name = (para.style.name or "") if para.style else ""
            lower_style = style_name.lower()

            if lower_style == "title":
                parts.append(f"\n# {text}")
            elif "heading" in lower_style:
                level = _heading_level(style_name)
                parts.append(f"\n{'#' * level} {text}")
            elif "list" in lower_style or _has_numbering(para):
                parts.append(f"  - {text}")
            else:
                parts.append(text)

        elif child in table_map:
            table = table_map[child]
            _format_table(table, parts)

    return "\n".join(parts)


def _heading_level(style_name: str) -> int:
    """Extract heading level from style name like 'Heading 2'. Defaults to 1."""
    for ch in style_name:
        if ch.isdigit():
            return min(int(ch), 6)
    return 1


def _has_numbering(para) -> bool:
    """Check if a paragraph has list-numbering XML properties."""
    pPr = para._element.pPr
    if pPr is None:
        return False
    for child in pPr:
        if child.tag.endswith("}numPr"):
            return True
    return False


def _format_table(table, parts: list[str]) -> None:
    """Render a DOCX table as a readable Markdown-style table."""
    parts.append("")  # blank line before table
    for row_idx, row in enumerate(table.rows):
        cells = [cell.text.strip().replace("\n", " ") for cell in row.cells]
        row_text = " | ".join(cells)
        parts.append(f"| {row_text} |")
        if row_idx == 0:
            parts.append("|" + " --- |" * len(cells))
    parts.append("")  # blank line after table

# PDF extraction
# ---------------------------------------------------------------------------

def _extract_from_pdf(file_bytes: bytes) -> str:
    """
    Extract text from a PDF.
    Tries pdfplumber first (better layout), falls back to PyMuPDF (fitz).
    """
    text = _extract_pdf_pdfplumber(file_bytes)
    if text and text.strip():
        return text

    logger.warning("pdfplumber returned empty text; trying PyMuPDF fallback.")
    text = _extract_pdf_pymupdf(file_bytes)
    if text and text.strip():
        return text

    raise ValueError(
        "Could not extract any text from the PDF. "
        "The file may be scanned/image-based. Please use a text-selectable PDF."
    )


def _extract_pdf_pdfplumber(file_bytes: bytes) -> str:
    try:
        import pdfplumber
    except ImportError:
        logger.warning("pdfplumber not installed; skipping. Run: pip install pdfplumber")
        return ""

    pages: list[str] = []
    with pdfplumber.open(io.BytesIO(file_bytes)) as pdf:
        for page in pdf.pages:
            page_text = page.extract_text()
            if page_text:
                pages.append(page_text.strip())
    return "\n\n".join(pages)


def _extract_pdf_pymupdf(file_bytes: bytes) -> str:
    try:
        import fitz  # PyMuPDF
    except ImportError:
        logger.warning("PyMuPDF not installed; skipping. Run: pip install PyMuPDF")
        return ""

    doc = fitz.open(stream=file_bytes, filetype="pdf")
    pages: list[str] = []
    for page in doc:
        pages.append(page.get_text())
    doc.close()
    return "\n\n".join(pages)

def _calculate_cost(model: str, input_tokens: int, output_tokens: int) -> dict[str, Any]:
    pricing = MODEL_PRICING.get(model, MODEL_PRICING["gpt-4o-mini"])
    input_cost = (input_tokens / 1000) * pricing["input"]
    output_cost = (output_tokens / 1000) * pricing["output"]
    return {
        "input_tokens": input_tokens,
        "output_tokens": output_tokens,
        "total_tokens": input_tokens + output_tokens,
        "input_cost_usd": round(input_cost, 6),
        "output_cost_usd": round(output_cost, 6),
        "total_cost_usd": round(input_cost + output_cost, 6),
        "model": model,
    }


def _merge_usage(usages: list[dict[str, Any]]) -> dict[str, Any]:
    """Merge multiple usage dicts from sequential or parallel LLM calls."""
    total_input = sum(u.get("input_tokens", 0) for u in usages)
    total_output = sum(u.get("output_tokens", 0) for u in usages)
    model = usages[0]["model"] if usages else LLM_MODEL
    return _calculate_cost(model, total_input, total_output)

# HEADING EXTRACTOR
# Heuristic pre-pass run BEFORE Phase 0. Extracts only structural markers
# (headings, numbered sections, underlined titles) from the document so Phase 0
# never needs to see the full body text. Cuts Phase 0 input tokens by ~80-90%.
# ═══════════════════════════════════════════════════════════════════════════

def _extract_document_headings(document_text: str) -> str:
    """
    Heuristically extract section headings and titles from a document.

    Detects (in priority order):
      1. Markdown headings       — lines starting with #, ##, ###, ####
      2. Numbered sections       — 1. Title / 1.1 Title / 1.1.1. Title
      3. Underlined titles       — RST / plain-text style (next line is === or ---)
      4. Bold-markdown headings  — **Heading** or __Heading__ alone on a line
      5. ALL-CAPS titles         — >=10 chars, uppercase letters present
      6. Standalone short lines  — <=80 chars, not ending in sentence punctuation,
                                   followed by a blank line

    Falls back to the first 2 000 chars of the document if nothing is found.
    The returned string is passed to Phase 0 instead of the full document.
    """
    lines = document_text.splitlines()
    n = len(lines)
    headings: list[str] = []
    skip_next = False

    for i, line in enumerate(lines):
        if skip_next:
            skip_next = False
            continue

        stripped = line.strip()
        if not stripped:
            continue

        # 1. Markdown headings: # / ## / ### / ####
        if re.match(r"^#{1,6}\s+\S", stripped):
            headings.append(stripped)
            continue

        # 2. Numbered section headings: 1., 1.1., 1.1.1., "Chapter 2", "Section 3"
        if re.match(r"^\d+(\.|\d+)*\.?\s+[A-Z\u0080-\uFFFF]", stripped) or \
                re.match(r"^(Chapter|Section|Part|Module)\s+\d+", stripped, re.IGNORECASE):
            headings.append(stripped)
            continue

        # 3. Underlined titles (RST/plain-text): next line is entirely === or ---
        if i + 1 < n:
            next_stripped = lines[i + 1].strip()
            if (
                next_stripped
                and re.match(r"^[=\-]{3,}$", next_stripped)
                and len(next_stripped) >= max(3, len(stripped) // 2)
            ):
                headings.append(stripped)
                skip_next = True
                continue

        # 4. Bold markdown alone on a line: **Heading** or __Heading__
        if re.match(r"^(\*\*|__).{3,}(\*\*|__)$", stripped):
            headings.append(stripped)
            continue

        # 5. ALL-CAPS line (min 10 chars, contains actual letters, not a divider)
        if (
            len(stripped) >= 10
            and stripped.upper() == stripped
            and re.search(r"[A-Z]{3,}", stripped)
            and not re.fullmatch(r"[^a-zA-Z0-9]*", stripped)  # not pure punctuation/symbols
        ):
            headings.append(stripped)
            continue

        # 6. Short standalone line followed by blank line (likely a plain-text title)
        if (
            2 <= len(stripped.split()) <= 12
            and len(stripped) <= 80
            and not stripped[-1] in (".", ",", ";", ":")
            and not re.search(r"\w\.\s+\w", stripped)  # not mid-sentence period
            and i + 1 < n
            and not lines[i + 1].strip()
        ):
            headings.append(stripped)
            continue

    if headings:
        return "\n".join(headings)

    # Fallback: the LLM will receive the first 2 000 chars if no headings found
    logger.warning(
        "_extract_document_headings: no headings detected; falling back to first 2000 chars."
    )
    return document_text[:2000]

def _slice_module_text(document_text: str, module_heading: str, next_heading: str | None) -> str:
    """
    Extract the text slice for a module by finding its heading in the document.
    Returns from module_heading up to (but not including) next_heading, or end of doc.
    Falls back to full document if heading is not found.
    """
    start_idx = document_text.find(module_heading)
    if start_idx == -1:
        # Heading not found verbatim — try case-insensitive
        lower_doc = document_text.lower()
        lower_heading = module_heading.lower()
        start_idx = lower_doc.find(lower_heading)
    if start_idx == -1:
        return document_text  # fallback: full document

    if next_heading:
        end_idx = document_text.find(next_heading, start_idx + len(module_heading))
        if end_idx == -1:
            lower_doc = document_text.lower()
            end_idx = lower_doc.find(next_heading.lower(), start_idx + len(module_heading))
        if end_idx == -1:
            return document_text[start_idx:]
        return document_text[start_idx:end_idx]

    return document_text[start_idx:]

# PHASE 2 — PYTHON INTEGRATION (no extra LLM call)
# Merges all per-module outputs, deduplicates shared data, cross-validates
# opens_screen references across modules.
# ═══════════════════════════════════════════════════════════════════════════

def _integrate_module_results(module_results: list[dict[str, Any]]) -> dict[str, Any]:
    """
    Merge all per-module extraction results into one final JSON.
    Deduplicates enums and business rules. Validates opens_screen references.
    """
    all_modules: list[dict[str, Any]] = []
    all_enums: dict[str, set] = {}         # enum name → set of values
    all_rules: list[str] = []
    all_workflows: list[dict[str, Any]] = []
    seen_rules: set[str] = set()

    for result in module_results:
        module = result.get("module", {})
        if module:
            all_modules.append(module)

        # Deduplicate enums — merge values from same enum name across modules
        for enum in result.get("shared_enums", []):
            name = enum.get("name", "")
            if name:
                if name not in all_enums:
                    all_enums[name] = set()
                all_enums[name].update(enum.get("values", []))

        # Deduplicate business rules (exact-match dedup)
        for rule in result.get("shared_business_rules", []):
            if rule and rule not in seen_rules:
                all_rules.append(rule)
                seen_rules.add(rule)

        all_workflows.extend(result.get("shared_workflows", []))

    # Build flat set of all defined screen names for cross-reference validation
    all_screen_names: set[str] = set()
    for mod in all_modules:
        for screen in mod.get("screens", []):
            name = screen.get("name", "")
            if name:
                all_screen_names.add(name)

    # Log any opens_screen references that point to undefined screens
    for mod in all_modules:
        for screen in mod.get("screens", []):
            _validate_opens_screen_refs(screen, all_screen_names)

    return {
        "modules": all_modules,
        "shared": {
            "enums": [
                {"name": name, "values": sorted(values)}
                for name, values in sorted(all_enums.items())
            ],
            "global_business_rules": all_rules,
            "workflows": all_workflows,
        },
    }


def _validate_opens_screen_refs(screen: dict[str, Any], known_screens: set[str]) -> None:
    """Walk a screen's components and log any opens_screen pointing to unknown screens."""
    for component in screen.get("components", []):
        comp_type = component.get("type", "")
        if comp_type == "toolbar":
            for action in component.get("actions", []):
                ref = action.get("opens_screen")
                if ref and ref not in known_screens:
                    logger.warning(
                        "opens_screen '%s' in toolbar action '%s' of screen '%s' "
                        "does not match any defined screen.",
                        ref, action.get("label"), screen.get("name"),
                    )
        elif comp_type == "grid":
            for ra in component.get("row_actions", []):
                ref = ra.get("opens_screen")
                if ref and ref not in known_screens:
                    logger.warning(
                        "opens_screen '%s' in row_action '%s' of screen '%s' "
                        "does not match any defined screen.",
                        ref, ra.get("label"), screen.get("name"),
                    )
        elif comp_type == "tabs":
            for tab in component.get("children", []):
                for child_comp in tab.get("components", []):
                    _validate_opens_screen_refs(
                        {"name": screen.get("name"), "components": [child_comp]},
                        known_screens,
                    )

async def extract_requirements_from_text(user_story_text: str) -> dict[str, Any]:
    """
    Three-phase extraction:
      Phase 0 — Segmentation : identify top-level modules from the document.
      Phase 1 — Per-module   : parallel LLM calls, one per module, each with
                               only that module's text slice. Combined
                               discovery + extraction in one focused call.
      Phase 2 — Integration  : pure-Python merge, dedup, cross-validate.
    """
    usages: list[dict[str, Any]] = []

    # ── Phase 0: Segment the document into modules ──────────────────────
    logger.info("Phase 0: Extracting headings then segmenting into modules...")
    headings_text = _extract_document_headings(user_story_text)
    logger.info(
        "Phase 0: Heading extraction reduced input from %d → %d chars (%.0f%% saving).",
        len(user_story_text), len(headings_text),
        (1 - len(headings_text) / max(len(user_story_text), 1)) * 100,
    )
    seg_prompt = SEGMENTATION_USER_TEMPLATE.format(headings_text=headings_text)
    seg_result, seg_usages = await _call_openai_validated_json(
      system_prompt=SEGMENTATION_SYSTEM_PROMPT,
      user_prompt=seg_prompt,
      max_tokens=SEGMENTATION_MAX_TOKENS,
      schema=SEGMENTATION_SCHEMA,
      schema_name="segmentation",
    )
    usages.extend(seg_usages)
    modules_meta: list[dict[str, str]] = seg_result.get("modules", [])

    if not modules_meta:
        # Fallback: treat the whole document as one module
        modules_meta = [{"name": "Application", "heading": "", "description": "Full document"}]

    logger.info("Segmentation identified %d module(s): %s",
                len(modules_meta), [m["name"] for m in modules_meta])

    # ── Phase 1: Per-module extraction (all in parallel) ─────────────────
    logger.info("Phase 1: Extracting %d module(s) in parallel...", len(modules_meta))

    async def extract_module(idx: int, module_meta: dict[str, str]) -> dict[str, Any]:
        heading = module_meta.get("heading", "")
        next_heading = modules_meta[idx + 1].get("heading", "") if idx + 1 < len(modules_meta) else None
        module_text = _slice_module_text(user_story_text, heading, next_heading)

        prompt = MODULE_EXTRACTION_USER_TEMPLATE.format(module_text=module_text)
        result, module_usages = await _call_openai_validated_json(
          system_prompt=MODULE_EXTRACTION_SYSTEM_PROMPT,
          user_prompt=prompt,
          max_tokens=MODULE_MAX_TOKENS,
          schema=MODULE_OUTPUT_SCHEMA,
          schema_name=f"module_output:{module_meta.get('name', 'unknown')}",
        )
        usages.extend(module_usages)

        # Ensure the module name from segmentation is used if LLM left it blank
        if result.get("module") and not result["module"].get("name"):
            result["module"]["name"] = module_meta["name"]

        screen_count = len(result.get("module", {}).get("screens", []))
        logger.info("Module '%s' → %d screens extracted.", module_meta["name"], screen_count)
        return result

    module_results: list[dict[str, Any]] = await asyncio.gather(
        *[extract_module(i, m) for i, m in enumerate(modules_meta)]
    )

    # ── Phase 2: Integrate all module results ────────────────────────────
    logger.info("Phase 2: Integrating %d module results...", len(module_results))
    requirements = _integrate_module_results(list(module_results))
    _validate_json_schema(requirements, FINAL_REQUIREMENTS_SCHEMA, "final_requirements")

    total_screens = sum(
        len(mod.get("screens", [])) for mod in requirements.get("modules", [])
    )
    logger.info("Integration complete. Total screens: %d", total_screens)

    return {
        "requirements": requirements,
        "usage": _merge_usage(usages),
    }

# OPENAI ADAPTER
# ═══════════════════════════════════════════════════════════════════════════

async def _call_openai(
    system_prompt: str,
    user_prompt: str,
    max_tokens: int = 16384,
) -> tuple[str, dict[str, Any]]:
    """Call OpenAI Chat Completions API with given system + user prompts."""
    if not LLM_API_KEY:
        raise RuntimeError("OPENAI_API_KEY is not set.")

    url = "https://api.openai.com/v1/chat/completions"
    headers = {
        "Authorization": f"Bearer {LLM_API_KEY}",
        "Content-Type": "application/json",
    }
    payload = {
        "model": LLM_MODEL,
        "messages": [
            {"role": "system", "content": system_prompt},
            {"role": "user",   "content": user_prompt},
        ],
        "temperature": 0.1,
        "max_tokens": max_tokens,
        "response_format": {"type": "json_object"},
    }

    async with httpx.AsyncClient(timeout=LLM_TIMEOUT) as client:
        resp = await client.post(url, json=payload, headers=headers)
        resp.raise_for_status()
        data = resp.json()

    content = data["choices"][0]["message"]["content"]

    raw_usage = data.get("usage", {})
    usage = _calculate_cost(
        LLM_MODEL,
        raw_usage.get("prompt_tokens", 0),
        raw_usage.get("completion_tokens", 0),
    )
    return content, usage


# ═══════════════════════════════════════════════════════════════════════════
# JSON PARSING
# ═══════════════════════════════════════════════════════════════════════════

def _parse_json_response(raw: str) -> dict[str, Any]:
    """Parse LLM response to a Python dict, stripping markdown fences if present."""
    if not raw or not raw.strip():
        raise RuntimeError("LLM returned an empty response.")

    cleaned = re.sub(r"^```(?:json)?\s*", "", raw.strip(), flags=re.IGNORECASE)
    cleaned = re.sub(r"\s*```$", "", cleaned.strip())

    try:
        return json.loads(cleaned)
    except json.JSONDecodeError as e:
        logger.error("LLM response was not valid JSON:\n%s", raw[:500])
        raise RuntimeError(
            f"LLM did not return valid JSON. Parse error: {e}. "
            f"Raw response (first 300 chars): {raw[:300]}"
        )


def _validate_json_schema(data: dict[str, Any], schema: dict[str, Any], schema_name: str) -> None:
    """Validate parsed JSON payload against schema and raise a readable error if invalid."""
    validator = Draft202012Validator(schema)
    errors = sorted(validator.iter_errors(data), key=lambda e: e.path)
    if not errors:
        return

    first = errors[0]
    path = ".".join(str(p) for p in first.path) or "<root>"
    raise RuntimeError(
        f"Schema validation failed for {schema_name} at '{path}': {first.message}"
    )


async def _call_openai_validated_json(
    system_prompt: str,
    user_prompt: str,
    max_tokens: int,
    schema: dict[str, Any],
    schema_name: str,
) -> tuple[dict[str, Any], list[dict[str, Any]]]:
    """
    Call OpenAI, parse JSON, validate against schema, and retry if invalid.
    Returns (validated_json, [usage_per_attempt]).
    """
    attempt = 0
    usages: list[dict[str, Any]] = []
    last_error: Exception | None = None
    prompt = user_prompt

    while attempt <= SCHEMA_VALIDATION_RETRIES:
        attempt += 1
        raw, usage = await _call_openai(system_prompt, prompt, max_tokens=max_tokens)
        usages.append(usage)

        try:
            parsed = _parse_json_response(raw)
            _validate_json_schema(parsed, schema, schema_name)
            return parsed, usages
        except Exception as exc:
            last_error = exc
            if attempt > SCHEMA_VALIDATION_RETRIES:
                break

            logger.warning(
                "Attempt %d/%d failed for %s: %s",
                attempt,
                SCHEMA_VALIDATION_RETRIES + 1,
                schema_name,
                exc,
            )

            prompt = (
                f"{user_prompt}\n\n"
                "IMPORTANT: Your previous JSON was invalid. Regenerate and return ONLY valid JSON "
                f"that satisfies the required shape. Error: {exc}"
            )

    raise RuntimeError(
        f"LLM output could not be validated for {schema_name} after "
        f"{SCHEMA_VALIDATION_RETRIES + 1} attempts: {last_error}"
    )

"""
User Story Parser Router
Handles file upload (docx/pdf), text extraction, and LLM-based requirement extraction.
"""

import logging
from fastapi import APIRouter, File, UploadFile, HTTPException, status
from fastapi.responses import JSONResponse

from user_story_parser.extractor import extract_text_from_file
from user_story_parser.llm_service import extract_requirements_from_text
from user_story_parser.validators import validate_uploaded_file
from db.sqlite_store import save_extraction

logger = logging.getLogger(__name__)

router = APIRouter(prefix="/user-story", tags=["User Story Parser"])


@router.post(
    "/parse",
    summary="Parse a user story document and extract structured requirements",
    description=(
        "Upload a user story document (.docx or .pdf). "
        "The API extracts the text, passes it through an LLM with a structured prompt, "
        "and returns a detailed JSON describing screens, components, fields, validations, "
        "business rules, and more."
    ),
    response_description="Structured JSON representation of the user story requirements",
)

async def parse_user_story(
    file: UploadFile = File(..., description="User story document (.docx or .pdf)"),
):
    # Step 1: Validate
    validate_uploaded_file(file)
 
    # Step 2: Read file bytes
    try:
        file_bytes = await file.read()
    except Exception as e:
        logger.error("Failed to read uploaded file: %s", e)
        raise HTTPException(
            status_code=status.HTTP_400_BAD_REQUEST,
            detail=f"Could not read uploaded file: {str(e)}",
        )
 
    # Step 3: Extract text
    try:
        extracted_text = extract_text_from_file(
            file_bytes=file_bytes,
            filename=file.filename,
            content_type=file.content_type,
        )
    except ValueError as e:
        raise HTTPException(status_code=status.HTTP_422_UNPROCESSABLE_ENTITY, detail=str(e))
    except Exception as e:
        logger.error("Text extraction failed: %s", e)
        raise HTTPException(
            status_code=status.HTTP_500_INTERNAL_SERVER_ERROR,
            detail=f"Text extraction failed: {str(e)}",
        )
 
    if not extracted_text or not extracted_text.strip():
        raise HTTPException(
            status_code=status.HTTP_422_UNPROCESSABLE_ENTITY,
            detail="No text could be extracted from the uploaded document.",
        )
 
    # Step 4: LLM extraction
    try:
        result = await extract_requirements_from_text(extracted_text)
    except Exception as e:
        logger.error("LLM extraction failed: %s: %s", type(e).__name__, e)
        raise HTTPException(
            status_code=status.HTTP_500_INTERNAL_SERVER_ERROR,
            detail=f"Requirement extraction via LLM failed ({type(e).__name__}): {str(e)}",
        )
 
    # Step 5: Persist extraction to SQLite
    extraction_id = save_extraction(
        user_story_id=file.filename,
        extracted_requirements=result["requirements"],
    )

    # Step 6: Return requirements + usage + cost
    return JSONResponse(
        content={
            "status": "success",
            "extraction_id": extraction_id,
            "filename": file.filename,
            "requirements": result["requirements"],
            "usage": result["usage"],
        }
    )

validator.py
"""
File validation utilities for uploaded user story documents.
"""

from fastapi import HTTPException, UploadFile, status

# 20 MB limit
MAX_FILE_SIZE_BYTES = 20 * 1024 * 1024

ALLOWED_EXTENSIONS = {".docx", ".pdf"}

ALLOWED_CONTENT_TYPES = {
    "application/pdf",
    "application/vnd.openxmlformats-officedocument.wordprocessingml.document",
    "application/octet-stream",  # some browsers send this for .docx
}


def validate_uploaded_file(file: UploadFile) -> None:
    """
    Validate the uploaded file for type and basic sanity.

    Raises:
        HTTPException 400 if validation fails.
    """
    if not file.filename:
        raise HTTPException(
            status_code=status.HTTP_400_BAD_REQUEST,
            detail="No filename provided.",
        )

    from pathlib import Path
    ext = Path(file.filename).suffix.lower()

    if ext not in ALLOWED_EXTENSIONS:
        raise HTTPException(
            status_code=status.HTTP_400_BAD_REQUEST,
            detail=(
                f"Unsupported file type '{ext}'. "
                f"Allowed types: {', '.join(sorted(ALLOWED_EXTENSIONS))}"
            ),
        )

    # Content-type check (lenient — browsers can send generic types)
    content_type = (file.content_type or "").lower()
    if content_type and content_type not in ALLOWED_CONTENT_TYPES:
        # Warn but don't hard-block — some valid .docx come as octet-stream
        import logging
        logging.getLogger(__name__).warning(
            "Unexpected content_type '%s' for file '%s'. Proceeding by extension.",
            content_type, file.filename,
        )

"""
planner_agent/models.py
────────────────────────
Pydantic models for the Planner Agent (schema v2.0).

Design philosophy: module → screens → components → interactions → files
This mirrors the extractor output structure so no intelligence is lost
and the code-generator stage can operate on one component at a time.
"""

from __future__ import annotations

from typing import Dict, List, Literal, Optional, Tuple
from pydantic import BaseModel, Field


# ─── Input: slim module for validation of raw-list input ──────────────────────

class ExtractedModule(BaseModel):
    """
    Thin validator used when run_planner receives a raw list of dicts.
    The real planner input is the full extraction dict {modules, shared}.
    """
    name: str
    description: str = ""
    screens: list = Field(default_factory=list)


# ─── Output: Component-level ──────────────────────────────────────────────────

class ActionPlan(BaseModel):
    """An actionable button / icon inside a toolbar or form."""
    name: str
    type: str = ""                        # primary | secondary | icon
    behavior: str = ""                    # open_modal | export_table | refresh | submit | delete | etc.
    opens_component: Optional[str] = None # target screen/component name (if behavior opens something)


class ComponentPlan(BaseModel):
    """
    Plan for one UI component on a screen.
    type drives which toolkit_mapping and which extra fields are populated.
    """
    component_name: str                   # e.g. ProcessToolbar, ProcessGrid, AddEditProcessForm
    type: str                             # toolbar | grid | form | filter_panel | tabs | info_panel
    toolkit_mapping: str                  # ConfigurableDashboard | ConfigurableForm | Button | Table | …
    similarity_query: str                 # component-level query for vector search (precise)

    # toolbar-specific
    actions: List[ActionPlan] = Field(default_factory=list)

    # grid-specific
    columns: List[str] = Field(default_factory=list)

    # form-specific
    field_names: List[str] = Field(default_factory=list)
    validations: List[str] = Field(default_factory=list)

    # data-access
    data_hook: str = ""                   # useApiRequest | createApiSlice | ""


# ─── Output: Screen-level ─────────────────────────────────────────────────────

class UserInteraction(BaseModel):
    """One end-to-end user-initiated workflow within a screen."""
    action: str                           # e.g. "Add New Process"
    flow: List[str]                       # ordered list of steps


class StateItem(BaseModel):
    """One piece of client-side React state in a screen."""
    name: str                             # camelCase variable name
    type: str                             # string | number | boolean | list | object | null


class ApiCall(BaseModel):
    """One REST API call required by a screen."""
    name: str                             # camelCase function name  e.g. getProcesses
    method: str                           # GET | POST | PUT | DELETE
    endpoint: str = ""                   # REST path  e.g. /api/production-processes
    hook: str = "useApiRequest"          # useApiRequest | createApiSlice


class DataFlow(BaseModel):
    """Client-side state + API call catalogue for one screen."""
    state: List[StateItem] = Field(default_factory=list)
    api_calls: List[ApiCall] = Field(default_factory=list)


class ScreenPlan(BaseModel):
    """Complete plan for one screen (dashboard, form, detail view, popup, etc.)."""
    screen_name: str
    type: str                             # dashboard | form | detail | popup
    route: str = ""                      # URL route  e.g. /production-process-master
    opens_as: str = "page"              # page | modal | popup
    similarity_query: str                 # screen-level query for vector search

    components: List[ComponentPlan] = Field(default_factory=list)
    user_interactions: List[UserInteraction] = Field(default_factory=list)
    data_flow: DataFlow = Field(default_factory=DataFlow)


# ─── Output: Module-level ─────────────────────────────────────────────────────

FileType = Literal["page", "component", "form", "hook", "service", "types", "styles"]


class SharedComponent(BaseModel):
    """A component reused across multiple screens within the same module."""
    name: str
    toolkit_mapping: str = ""
    used_in_screens: List[str] = Field(default_factory=list)


class FilePlan(BaseModel):
    """Plan for one source file in the module."""
    path: str                            # e.g. src/modules/ProductionProcessMaster/pages/...tsx
    type: FileType
    description: str
    uses_components: List[str] = Field(default_factory=list)  # names of ComponentPlan items
    toolkit_imports: Dict[str, List[str]] = Field(default_factory=dict)
    key_exports: List[str] = Field(default_factory=list)


class ModulePlan(BaseModel):
    """Complete plan for one UI module."""
    module_name: str
    description: str
    similarity_query: str                 # broad module-level query for retrieval

    screens: List[ScreenPlan]
    shared_components: List[SharedComponent] = Field(default_factory=list)
    file_structure: List[FilePlan] = Field(default_factory=list)


# ─── Top-level output ─────────────────────────────────────────────────────────

class LLMUsage(BaseModel):
    """Token usage and cost for the planner LLM call."""
    input_tokens: int = 0
    output_tokens: int = 0
    total_tokens: int = 0
    input_cost_usd: float = 0.0
    output_cost_usd: float = 0.0
    total_cost_usd: float = 0.0
    model: str = ""


class PlannerOutput(BaseModel):
    """Top-level output of the Planner Agent."""
    modules: List[ModulePlan]
    usage: LLMUsage = Field(default_factory=LLMUsage)

    def get_similarity_queries(self) -> Dict[str, List[str]]:
        """
        Returns {module_name: [module_query, screen_queries..., component_queries...]}
        for use by the retrieval stage.  All queries are labelled for context.
        """
        result: Dict[str, List[str]] = {}
        for m in self.modules:
            queries: List[str] = [f"[module] {m.similarity_query}"]
            for s in m.screens:
                queries.append(f"[{s.screen_name}] {s.similarity_query}")
                for c in s.components:
                    queries.append(f"[{c.component_name}] {c.similarity_query}")
            result[m.module_name] = queries
        return result

    def get_all_files(self) -> List[Tuple[str, FilePlan]]:
        """Returns [(module_name, FilePlan), ...] for every file across all modules."""
        return [(m.module_name, f) for m in self.modules for f in m.file_structure]



planner_agent.py
"""
agents/planner_agent.py
────────────────────────
LangGraph-based Planner Agent.

Flow:
  [START]
     │
     ▼
  toon_convert        ← compresses the rich extractor JSON to TOON
     │
     ▼
  llm_plan            ← calls GPT with toolkit-aware system prompt
     │
     ▼
  parse_and_validate  ← parses JSON, validates 3-level schema
     │
     ▼
  [END]  → PlannerOutput

The graph is intentionally linear (no cycles) because planning is a
single-shot operation.
"""

from __future__ import annotations

import json
import logging
from typing import Any, TypedDict

from jsonschema import Draft202012Validator

from langchain_core.messages import HumanMessage, SystemMessage
from langchain_openai import ChatOpenAI
from langgraph.graph import END, START, StateGraph

from planner_agent.toon_serializer import toon_rich_extraction
from planner_agent.planner_prompts import build_human_prompt, build_system_prompt

logger = logging.getLogger(__name__)


# ─── Output JSON Schema (mirrors Pydantic models in models.py) ────────────────
# Used by parse_and_validate to give structured, path-precise error messages
# the same way llm_service.py validates the requirement extractor output.

PLANNER_OUTPUT_SCHEMA: dict[str, Any] = {
    "type": "array",
    "items": {
        "type": "object",
        "required": ["module_name", "description", "similarity_query", "screens", "file_structure"],
        "additionalProperties": True,
        "properties": {
            "module_name":       {"type": "string", "minLength": 1},
            "description":       {"type": "string"},
            "similarity_query":  {"type": "string", "minLength": 1},
            "screens": {
                "type": "array",
                "minItems": 1,
                "items": {
                    "type": "object",
                    "required": ["screen_name", "type", "opens_as", "similarity_query", "components"],
                    "additionalProperties": True,
                    "properties": {
                        "screen_name":      {"type": "string", "minLength": 1},
                        "type":             {"type": "string", "enum": ["dashboard", "form", "detail", "popup"]},
                        "opens_as":         {"type": "string"},
                        "similarity_query": {"type": "string", "minLength": 1},
                        "components": {
                            "type": "array",
                            "minItems": 1,
                            "items": {
                                "type": "object",
                                "required": ["component_name", "type", "toolkit_mapping", "similarity_query"],
                                "additionalProperties": True,
                                "properties": {
                                    "component_name":   {"type": "string", "minLength": 1},
                                    "type":             {"type": "string"},
                                    "toolkit_mapping":  {"type": "string"},
                                    "similarity_query": {"type": "string", "minLength": 1},
                                },
                            },
                        },
                    },
                },
            },
            "file_structure": {
                "type": "array",
                "minItems": 1,
                "items": {
                    "type": "object",
                    "required": ["path", "type", "description"],
                    "additionalProperties": True,
                    "properties": {
                        "path":        {"type": "string"},
                        "type":        {"type": "string"},
                        "description": {"type": "string"},
                    },
                },
            },
        },
    },
}


# ─── Graph state ──────────────────────────────────────────────────────────────

class PlannerState(TypedDict, total=False):
    # Input
    extraction: dict              # full extractor output: {modules: [...], shared: {...}}

    # Intermediate
    toon_str: str                 # TOON-compressed extraction
    raw_llm_response: str         # raw string from the LLM
    raw_usage: dict               # token counts from the LLM response

    # Output
    manifest_plan: list[dict]     # parsed module plan list
    error: str                    # set if any step fails


# ─── Node: TOON conversion ────────────────────────────────────────────────────

def toon_convert(state: PlannerState) -> PlannerState:
    """
    Convert the full extraction dict (modules + shared) into compact TOON
    so the LLM receives a rich but token-efficient context.
    """
    extraction: dict = state["extraction"]
    toon = toon_rich_extraction(extraction)
    logger.info("[planner] TOON conversion done. Length: %d chars", len(toon))
    return {**state, "toon_str": toon}


# ─── Node: LLM planning ───────────────────────────────────────────────────────

def llm_plan(state: PlannerState, llm: ChatOpenAI) -> PlannerState:
    """
    Send the TOON-compressed requirements to the LLM with the
    toolkit-aware system prompt and get back the rich file manifest plan.
    Captures token usage from the response metadata.
    """
    messages = [
        SystemMessage(content=build_system_prompt()),
        HumanMessage(content=build_human_prompt(state["toon_str"])),
    ]

    logger.info("[planner] Calling LLM for rich file manifest plan …")
    response = llm.invoke(messages)
    raw = response.content.strip()
    logger.info("[planner] LLM responded (%d chars)", len(raw))

    # ── Extract token usage ────────────────────────────────────────────────
    raw_usage: dict = {}
    if hasattr(response, "usage_metadata") and response.usage_metadata:
        um = response.usage_metadata
        if isinstance(um, dict):
            raw_usage = {
                "input_tokens":  um.get("input_tokens", 0),
                "output_tokens": um.get("output_tokens", 0),
                "total_tokens":  um.get("total_tokens", 0),
            }
        else:
            raw_usage = {
                "input_tokens":  getattr(um, "input_tokens", 0),
                "output_tokens": getattr(um, "output_tokens", 0),
                "total_tokens":  getattr(um, "total_tokens", 0),
            }
    elif hasattr(response, "response_metadata"):
        token_usage = (response.response_metadata or {}).get("token_usage", {})
        raw_usage = {
            "input_tokens":  token_usage.get("prompt_tokens", 0),
            "output_tokens": token_usage.get("completion_tokens", 0),
            "total_tokens":  token_usage.get("total_tokens", 0),
        }

    return {**state, "raw_llm_response": raw, "raw_usage": raw_usage}


# ─── Node: Parse & validate ───────────────────────────────────────────────────

def parse_and_validate(state: PlannerState) -> PlannerState:
    """
    Parse the LLM's JSON output and validate the 3-level schema:
      module → screens → components
    Validation is non-fatal: returns parsed plan even with warnings so that
    downstream stages receive partial output rather than nothing.
    """
    raw = state.get("raw_llm_response", "")

    # Strip accidental markdown fences
    if raw.startswith("```"):
        raw = raw.split("```")[1]
        if raw.startswith("json"):
            raw = raw[4:]
        raw = raw.strip()

    try:
        plan: list[dict] = json.loads(raw)
    except json.JSONDecodeError as exc:
        err = f"JSON parse failed: {exc}"
        logger.error("[planner] %s", err)
        return {**state, "error": err}

    # ── JSON Schema validation (mirrors llm_service.py Draft202012Validator) ──
    validator = Draft202012Validator(PLANNER_OUTPUT_SCHEMA)
    errors = sorted(validator.iter_errors(plan), key=lambda e: list(e.path))
    if errors:
        issues = []
        for err in errors:
            path = ".".join(str(p) for p in err.path) or "<root>"
            issues.append(f"{path}: {err.message}")
        err_msg = "Schema validation issues: " + "; ".join(issues)
        logger.warning("[planner] %s", err_msg)
        # Non-fatal — return the plan so downstream stages get partial output
        return {**state, "manifest_plan": plan, "error": err_msg}

    logger.info("[planner] Plan validated. %d modules.", len(plan))
    return {**state, "manifest_plan": plan}


# ─── Graph factory ────────────────────────────────────────────────────────────

def build_planner_graph(llm: ChatOpenAI) -> Any:
    """
    Compile and return the LangGraph planner graph.

    Usage:
        graph = build_planner_graph(llm)
        result = graph.invoke({"extraction": {...}})
        plan = result["manifest_plan"]
    """
    graph = StateGraph(PlannerState)

    graph.add_node("toon_convert", toon_convert)
    graph.add_node("llm_plan", lambda s: llm_plan(s, llm))
    graph.add_node("parse_and_validate", parse_and_validate)

    graph.add_edge(START, "toon_convert")
    graph.add_edge("toon_convert", "llm_plan")
    graph.add_edge("llm_plan", "parse_and_validate")
    graph.add_edge("parse_and_validate", END)

    return graph.compile()

run_planner.py
"""
run_planner.py
──────────────
Public entry point for the Planner Agent stage.

Takes the full extraction output from the user_story_parser, runs it
through the LangGraph planner, and returns a structured PlannerOutput
with per-module, per-screen, and per-component plans plus token usage.

Usage (from code):
    from planner_agent.run_planner import run_planner
    result = run_planner(extraction)     # extraction = {"modules": [...], "shared": {...}}

Usage (CLI):
    python -m planner_agent.run_planner --input extraction.json --output manifest.json
"""

from __future__ import annotations

import json
import logging
import os
from pathlib import Path

from langchain_openai import ChatOpenAI

from planner_agent.planner_agent import build_planner_graph
from planner_agent.models import LLMUsage, ModulePlan, PlannerOutput

logging.basicConfig(level=logging.INFO, format="%(levelname)s | %(message)s")
logger = logging.getLogger(__name__)


# ─── Model configuration ──────────────────────────────────────────────────────

MODEL_NAME = "gpt-4.1-mini"

# Cost per 1 000 tokens (USD)
MODEL_PRICING: dict[str, dict[str, float]] = {
    "gpt-4.1-mini": {"input": 0.000400, "output": 0.001600},
    "gpt-4o-mini":  {"input": 0.000150, "output": 0.000600},
    "gpt-4o":       {"input": 0.002500, "output": 0.010000},
    "gpt-4.1":      {"input": 0.002000, "output": 0.008000},
}


# ─── Helpers ──────────────────────────────────────────────────────────────────

def _make_llm() -> ChatOpenAI:
    return ChatOpenAI(
        model=MODEL_NAME,
        temperature=0,
        api_key=os.environ["OPENAI_API_KEY"],
    )


def _calculate_usage(model: str, input_tokens: int, output_tokens: int) -> LLMUsage:
    pricing = MODEL_PRICING.get(model, MODEL_PRICING[MODEL_NAME])
    input_cost  = (input_tokens  / 1_000) * pricing["input"]
    output_cost = (output_tokens / 1_000) * pricing["output"]
    return LLMUsage(
        input_tokens=input_tokens,
        output_tokens=output_tokens,
        total_tokens=input_tokens + output_tokens,
        input_cost_usd=round(input_cost, 6),
        output_cost_usd=round(output_cost, 6),
        total_cost_usd=round(input_cost + output_cost, 6),
        model=model,
    )


# ─── Main runner ──────────────────────────────────────────────────────────────

def run_planner(extraction: dict) -> PlannerOutput:
    """
    Synchronous entry point for the Planner Agent.

    Args:
        extraction: Full extraction output from the user_story_parser stage:
                    {"modules": [...], "shared": {...}}

    Returns:
        PlannerOutput with a .modules list of ModulePlan objects (each with
        screens, components, similarity queries, file structure) and a
        .usage object with token counts and cost breakdown.
    """
    if not extraction.get("modules"):
        raise ValueError("extraction must contain a non-empty 'modules' list")

    llm   = _make_llm()
    graph = build_planner_graph(llm)

    logger.info("Running planner for %d modules …", len(extraction["modules"]))
    result = graph.invoke({"extraction": extraction})

    if result.get("error") and not result.get("manifest_plan"):
        raise RuntimeError(f"Planner failed: {result['error']}")

    raw_plans: list[dict] = result["manifest_plan"]
    plans = [ModulePlan(**p) for p in raw_plans]

    raw_usage = result.get("raw_usage") or {}
    usage = _calculate_usage(
        MODEL_NAME,
        raw_usage.get("input_tokens", 0),
        raw_usage.get("output_tokens", 0),
    )

    output = PlannerOutput(modules=plans, usage=usage)
    logger.info(
        "Planner complete. %d module plans. %d total tokens. Cost: $%.4f",
        len(plans), usage.total_tokens, usage.total_cost_usd,
    )
    return output


# ─── CLI ──────────────────────────────────────────────────────────────────────

def _cli() -> None:
    import argparse

    parser = argparse.ArgumentParser(description="Run the Planner Agent")
    parser.add_argument(
        "--input", required=True,
        help="Path to extraction JSON file (output of user_story_parser stage)",
    )
    parser.add_argument(
        "--output", default="manifest_plan.json",
        help="Path to write the manifest plan JSON",
    )
    parser.add_argument(
        "--print-queries", action="store_true",
        help="Print similarity queries after planning",
    )
    args = parser.parse_args()

    extraction = json.loads(Path(args.input).read_text())
    output = run_planner(extraction)

    plan_data = [p.model_dump() for p in output.modules]
    Path(args.output).write_text(json.dumps(plan_data, indent=2))
    logger.info("Manifest plan written to %s", args.output)

    print(f"\n=== Usage ({output.usage.model}) ===")
    print(f"  Input:  {output.usage.input_tokens:,} tokens  (${output.usage.input_cost_usd:.4f})")
    print(f"  Output: {output.usage.output_tokens:,} tokens  (${output.usage.output_cost_usd:.4f})")
    print(f"  Total:  {output.usage.total_tokens:,} tokens  (${output.usage.total_cost_usd:.4f})")

    if args.print_queries:
        print("\n=== Similarity Queries ===")
        for module, queries in output.get_similarity_queries().items():
            for q in queries:
                print(f"  [{module}]  {q}")


if __name__ == "__main__":
    _cli()

toon_serializer.py
"""
TOON — Token-Oriented Object Notation  (Spec v3.0)
────────────────────────────────────────────────────
Line-oriented, indentation-based encoding of the JSON data model.
Designed for LLM prompts — compact, human-readable, lossless round-trip.

Syntax overview (per spec):
  Objects   → key: value (indentation for nesting)
  Primitive arrays → key[N]: v1,v2,v3
  Tabular arrays   → key[N]{f1,f2}: \\n  row1 \\n  row2
  Mixed arrays     → key[N]: \\n  - item
  Booleans  → true / false
  Null      → null
  Strings   → quoted only when they contain :,[]{} or equal true/false/null

Reference: https://github.com/toon-format/spec/blob/main/SPEC.md
"""

from __future__ import annotations

import math
import re
from typing import Any

# ─── Quoting helpers ───────────────────────────────────────────────────────────

# Characters that force quoting (spec §7.2)
_NEEDS_QUOTE_RE = re.compile(r'[\[\]{}\:\,\\\"\n\r\t]')
_NUMERIC_RE = re.compile(r'^-?\d+(?:\.\d+)?(?:e[+-]?\d+)?$', re.IGNORECASE)
_LEADING_ZERO_RE = re.compile(r'^0\d+$')
_RESERVED = {"true", "false", "null"}


def _needs_quotes(value: str, delimiter: str = ",") -> bool:
    """Return True if a string value must be wrapped in quotes (§7.2)."""
    if value == "":
        return True
    if value != value.strip():  # leading / trailing whitespace
        return True
    if value in _RESERVED:
        return True
    if _NUMERIC_RE.match(value) or _LEADING_ZERO_RE.match(value):
        return True
    if _NEEDS_QUOTE_RE.search(value):
        return True
    if delimiter in value:
        return True
    if value.startswith("-"):
        return True
    return False


def _escape(value: str) -> str:
    """Escape a string for TOON quoted output (§7.1 — only 5 escapes)."""
    return (
        value
        .replace("\\", "\\\\")
        .replace('"', '\\"')
        .replace("\n", "\\n")
        .replace("\r", "\\r")
        .replace("\t", "\\t")
    )


def _quote(value: str, delimiter: str = ",") -> str:
    """Return the string quoted if necessary, else as-is."""
    if _needs_quotes(value, delimiter):
        return f'"{_escape(value)}"'
    return value


# ─── Canonical number formatting (§2) ──────────────────────────────────────────

def _format_number(n: int | float) -> str:
    """Canonical decimal — no exponent, no trailing zeros, -0→0."""
    if isinstance(n, float):
        if math.isnan(n) or math.isinf(n):
            return "null"
        if n == 0.0:
            return "0"
        # Avoid exponent notation
        formatted = f"{n:.17g}"
        if "e" in formatted or "E" in formatted:
            formatted = f"{n:.20f}".rstrip("0").rstrip(".")
        # Remove trailing zeros in fractional part
        if "." in formatted:
            formatted = formatted.rstrip("0").rstrip(".")
        return formatted
    # int
    return str(n)


# ─── Encoder ──────────────────────────────────────────────────────────────────

_INDENT = "  "  # 2 spaces per spec default


def _encode_value(val: Any, depth: int, delimiter: str = ",") -> list[str]:
    """Encode a JSON-model value into TOON lines at the given indentation depth."""
    if val is None:
        return ["null"]

    if isinstance(val, bool):
        return ["true" if val else "false"]

    if isinstance(val, (int, float)):
        return [_format_number(val)]

    if isinstance(val, str):
        return [_quote(val, delimiter)]

    if isinstance(val, list):
        return _encode_array(val, depth, delimiter)

    if isinstance(val, dict):
        return _encode_object(val, depth, delimiter)

    return [_quote(str(val), delimiter)]


def _is_all_primitives(lst: list) -> bool:
    """Check if every element is a primitive (string, number, bool, None)."""
    return all(isinstance(v, (str, int, float, bool, type(None))) for v in lst)


def _is_tabular(lst: list) -> bool:
    """Check if a list qualifies for tabular encoding (§9.3)."""
    if not lst:
        return False
    if not all(isinstance(v, dict) for v in lst):
        return False
    keys = set(lst[0].keys())
    for obj in lst[1:]:
        if set(obj.keys()) != keys:
            return False
    # All values must be primitives
    for obj in lst:
        for v in obj.values():
            if not isinstance(v, (str, int, float, bool, type(None))):
                return False
    return True


def _encode_primitive(val: Any, delimiter: str = ",") -> str:
    """Encode a single primitive to an inline string."""
    if val is None:
        return "null"
    if isinstance(val, bool):
        return "true" if val else "false"
    if isinstance(val, (int, float)):
        return _format_number(val)
    return _quote(str(val), delimiter)


def _encode_inline_array(lst: list, delimiter: str = ",") -> str:
    """Encode a primitive list as inline: [N]: v1,v2,v3"""
    parts = [_encode_primitive(v, delimiter) for v in lst]
    return f"[{len(lst)}]: {delimiter.join(parts)}" if lst else "[0]:"


def _encode_array(lst: list, depth: int, delimiter: str = ",") -> list[str]:
    """Encode an array — choose inline / tabular / expanded form."""
    if not lst:
        return ["[0]:"]

    # Inline primitive array (§9.1)
    if _is_all_primitives(lst):
        return [_encode_inline_array(lst, delimiter)]

    # Tabular form (§9.3)
    if _is_tabular(lst):
        return _encode_tabular(lst, depth, delimiter)

    # Expanded / mixed form (§9.4)
    return _encode_expanded(lst, depth, delimiter)


def _encode_tabular(lst: list[dict], depth: int, delimiter: str = ",") -> list[str]:
    """Encode a uniform list of primitive-only dicts as a tabular array."""
    fields = list(lst[0].keys())
    field_str = delimiter.join(_quote_key(f) for f in fields)
    header = f"[{len(lst)}]{{{field_str}}}:"
    lines = [header]
    indent = _INDENT * (depth + 1)
    for obj in lst:
        row_parts = [_encode_primitive(obj.get(f), delimiter) for f in fields]
        lines.append(f"{indent}{delimiter.join(row_parts)}")
    return lines


def _encode_expanded(lst: list, depth: int, delimiter: str = ",") -> list[str]:
    """Encode a mixed / non-uniform array as expanded list items (§9.4, §10)."""
    lines: list[str] = [f"[{len(lst)}]:"]
    indent = _INDENT * (depth + 1)

    for item in lst:
        if isinstance(item, dict):
            # Object as list item (§10) — first field on hyphen line
            lines.extend(_encode_list_item_object(item, depth + 1, delimiter))
        elif isinstance(item, list):
            # Nested array as list item
            inner = _encode_array(item, depth + 1, delimiter)
            lines.append(f"{indent}- {inner[0]}")
            lines.extend(inner[1:])
        else:
            lines.append(f"{indent}- {_encode_primitive(item, delimiter)}")

    return lines


def _encode_list_item_object(obj: dict, depth: int, delimiter: str = ",") -> list[str]:
    """Encode a dict as a list-item object (§10)."""
    indent = _INDENT * depth
    inner_indent = _INDENT * (depth + 1)
    lines: list[str] = []

    keys = list(obj.keys())
    if not keys:
        lines.append(f"{indent}-")
        return lines

    first_key = keys[0]
    first_val = obj[first_key]

    # First field on the hyphen line
    if isinstance(first_val, (str, int, float, bool, type(None))):
        lines.append(f"{indent}- {_quote_key(first_key)}: {_encode_primitive(first_val, delimiter)}")
    elif isinstance(first_val, list) and _is_all_primitives(first_val):
        lines.append(f"{indent}- {_quote_key(first_key)}{_encode_inline_array(first_val, delimiter)}")
    elif isinstance(first_val, list) and _is_tabular(first_val):
        tab_lines = _encode_tabular(first_val, depth + 1, delimiter)
        lines.append(f"{indent}- {_quote_key(first_key)}{tab_lines[0]}")
        lines.extend(tab_lines[1:])
    elif isinstance(first_val, list):
        arr_lines = _encode_array(first_val, depth + 1, delimiter)
        lines.append(f"{indent}- {_quote_key(first_key)}{arr_lines[0]}")
        lines.extend(arr_lines[1:])
    elif isinstance(first_val, dict):
        lines.append(f"{indent}- {_quote_key(first_key)}:")
        obj_lines = _encode_object_fields(first_val, depth + 2, delimiter)
        lines.extend(obj_lines)
    else:
        lines.append(f"{indent}- {_quote_key(first_key)}: {_encode_primitive(first_val, delimiter)}")

    # Remaining fields at depth +1
    for key in keys[1:]:
        val = obj[key]
        encoded = _encode_key_value(key, val, depth + 1, delimiter)
        for line in encoded:
            if not line.startswith(_INDENT * (depth + 1)):
                lines.append(f"{inner_indent}{line}")
            else:
                lines.append(line)

    return lines


def _quote_key(key: str) -> str:
    """Quote a key per §7.3 — unquoted if it matches ^[A-Za-z_][A-Za-z0-9_.]*$."""
    if re.match(r'^[A-Za-z_][A-Za-z0-9_.]*$', key):
        return key
    return f'"{_escape(key)}"'


def _encode_key_value(key: str, val: Any, depth: int, delimiter: str = ",") -> list[str]:
    """Encode a single key:value pair (may span multiple lines for nesting)."""
    indent = _INDENT * depth
    qk = _quote_key(key)

    if val is None:
        return [f"{indent}{qk}: null"]

    if isinstance(val, bool):
        return [f"{indent}{qk}: {'true' if val else 'false'}"]

    if isinstance(val, (int, float)):
        return [f"{indent}{qk}: {_format_number(val)}"]

    if isinstance(val, str):
        return [f"{indent}{qk}: {_quote(val, delimiter)}"]

    if isinstance(val, list):
        arr_lines = _encode_array(val, depth, delimiter)
        result = [f"{indent}{qk}{arr_lines[0]}"]
        result.extend(arr_lines[1:])
        return result

    if isinstance(val, dict):
        if not val:
            return [f"{indent}{qk}:"]
        lines = [f"{indent}{qk}:"]
        lines.extend(_encode_object_fields(val, depth + 1, delimiter))
        return lines

    return [f"{indent}{qk}: {_quote(str(val), delimiter)}"]


def _encode_object_fields(obj: dict, depth: int, delimiter: str = ",") -> list[str]:
    """Encode all fields of an object at the given depth."""
    lines: list[str] = []
    for k, v in obj.items():
        lines.extend(_encode_key_value(k, v, depth, delimiter))
    return lines


def _encode_object(obj: dict, depth: int, delimiter: str = ",") -> list[str]:
    """Encode a top-level or nested object."""
    if not obj:
        return []
    return _encode_object_fields(obj, depth, delimiter)


def to_toon(data: Any, indent_size: int = 2) -> str:
    """
    Convert a Python object into a TOON string (spec v3.0).

    Examples:
        {"name": "Ada", "active": true}
        → name: Ada
          active: true

        [{"id": 1, "name": "Alice"}, {"id": 2, "name": "Bob"}]
        → [2]{id,name}:
            1,Alice
            2,Bob
    """
    global _INDENT
    _INDENT = " " * indent_size

    if isinstance(data, dict):
        lines = _encode_object(data, depth=0)
    elif isinstance(data, list):
        lines = _encode_value(data, depth=0)
    else:
        lines = _encode_value(data, depth=0)

    return "\n".join(lines)


# ─── Decoder ──────────────────────────────────────────────────────────────────

class _ToonDecoder:
    """Decode a TOON string back into a Python object."""

    def __init__(self, text: str, indent_size: int = 2):
        self.lines: list[str] = text.split("\n")
        self.pos = 0
        self.indent_size = indent_size

    def decode(self) -> Any:
        # Skip blank lines
        self._skip_blank()
        if self.pos >= len(self.lines):
            return {}

        first_line = self.lines[self.pos].strip()

        # Root array header
        if re.match(r'^\[(\d+)', first_line):
            return self._parse_array_from_header(0)

        # Single primitive (exactly one non-empty line with no colon and no header)
        non_empty = [l for l in self.lines if l.strip()]
        if len(non_empty) == 1 and ":" not in first_line and not re.match(r'^\[', first_line):
            return self._parse_primitive_token(first_line)

        # Root object
        return self._parse_object(0)

    def _depth(self, line: str) -> int:
        stripped = line.lstrip(" ")
        spaces = len(line) - len(stripped)
        return spaces // self.indent_size

    def _skip_blank(self) -> None:
        while self.pos < len(self.lines) and not self.lines[self.pos].strip():
            self.pos += 1

    def _parse_object(self, min_depth: int) -> dict:
        result: dict = {}
        while self.pos < len(self.lines):
            self._skip_blank()
            if self.pos >= len(self.lines):
                break
            line = self.lines[self.pos]
            if not line.strip():
                self.pos += 1
                continue
            d = self._depth(line)
            if d < min_depth:
                break
            if d > min_depth:
                break

            stripped = line.strip()

            # key-value or key-array
            key, rest = self._split_key(stripped)
            if key is None:
                break

            if rest == "":
                # Nested object: key:
                self.pos += 1
                result[key] = self._parse_object(min_depth + 1)
            elif rest.startswith("["):
                # Array header
                result[key] = self._parse_array_value(rest, min_depth)
            else:
                # Primitive value
                result[key] = self._parse_primitive_token(rest)
                self.pos += 1

        return result

    def _split_key(self, stripped: str) -> tuple[str | None, str]:
        """Extract key and rest from a key: value line."""
        if stripped.startswith('"'):
            # Quoted key
            end = stripped.index('"', 1)
            while end > 0 and stripped[end - 1] == '\\':
                end = stripped.index('"', end + 1)
            key = self._unescape(stripped[1:end])
            after = stripped[end + 1:].strip()
            if after.startswith(":"):
                rest = after[1:].strip()
                return key, rest
            elif after.startswith("["):
                return key, after
            return None, ""
        else:
            # Unquoted key — find first colon or bracket
            bracket_pos = stripped.find("[")
            colon_pos = stripped.find(":")

            if bracket_pos >= 0 and (colon_pos < 0 or bracket_pos < colon_pos):
                key = stripped[:bracket_pos].strip()
                rest = stripped[bracket_pos:]
                return key, rest

            if colon_pos < 0:
                return None, ""
            key = stripped[:colon_pos].strip()
            rest = stripped[colon_pos + 1:].strip()
            return key, rest

    def _parse_array_value(self, rest: str, current_depth: int) -> Any:
        """Parse an array given the header portion starting with '['."""
        # Parse header: [N]{fields}: inline_values
        m = re.match(r'\[(\d+)\](\{[^}]*\})?:\s*(.*)', rest)
        if not m:
            self.pos += 1
            return []

        count = int(m.group(1))
        fields_str = m.group(2)
        inline = m.group(3).strip()

        if fields_str:
            # Tabular form
            fields = self._parse_field_names(fields_str[1:-1])
            self.pos += 1
            rows: list[dict] = []
            for _ in range(count):
                self._skip_blank()
                if self.pos >= len(self.lines):
                    break
                row_line = self.lines[self.pos].strip()
                values = self._split_delimited(row_line)
                obj = {}
                for fi, fname in enumerate(fields):
                    obj[fname] = self._parse_primitive_token(values[fi]) if fi < len(values) else None
                rows.append(obj)
                self.pos += 1
            return rows

        if inline:
            # Inline primitive array
            self.pos += 1
            values = self._split_delimited(inline)
            return [self._parse_primitive_token(v) for v in values]

        # Expanded list
        self.pos += 1
        items: list = []
        for _ in range(count):
            self._skip_blank()
            if self.pos >= len(self.lines):
                break
            line = self.lines[self.pos].strip()
            if line.startswith("- "):
                item_content = line[2:].strip()
                if item_content.startswith("["):
                    # Nested array
                    items.append(self._parse_array_value(item_content, current_depth + 1))
                elif ":" in item_content or item_content.startswith('"'):
                    # Object as list item
                    key, val_rest = self._split_key(item_content)
                    if key is not None:
                        obj = {}
                        if val_rest.startswith("["):
                            obj[key] = self._parse_array_value(val_rest, current_depth + 1)
                        elif val_rest == "":
                            self.pos += 1
                            obj[key] = self._parse_object(current_depth + 2)
                            # Parse remaining fields at depth+1
                            remaining = self._parse_object(current_depth + 1)
                            obj.update(remaining)
                            continue
                        else:
                            obj[key] = self._parse_primitive_token(val_rest)
                        self.pos += 1
                        remaining = self._parse_object(current_depth + 1)
                        obj.update(remaining)
                        items.append(obj)
                        continue
                    else:
                        items.append(self._parse_primitive_token(item_content))
                else:
                    items.append(self._parse_primitive_token(item_content))
                self.pos += 1
            elif line == "-":
                items.append({})
                self.pos += 1
            else:
                break
        return items

    def _parse_array_from_header(self, depth: int) -> Any:
        """Parse a root-level array."""
        line = self.lines[self.pos].strip()
        return self._parse_array_value(line, depth)

    def _parse_field_names(self, s: str) -> list[str]:
        """Parse comma-separated field names from a {f1,f2} segment."""
        return [f.strip().strip('"') for f in s.split(",")]

    def _split_delimited(self, s: str, delimiter: str = ",") -> list[str]:
        """Split a delimited string, respecting quoting."""
        parts: list[str] = []
        current: list[str] = []
        in_quotes = False
        i = 0
        while i < len(s):
            ch = s[i]
            if ch == '"' and not in_quotes:
                in_quotes = True
                current.append(ch)
            elif ch == '"' and in_quotes:
                if i > 0 and s[i - 1] == '\\':
                    current.append(ch)
                else:
                    in_quotes = False
                    current.append(ch)
            elif ch == delimiter and not in_quotes:
                parts.append("".join(current).strip())
                current = []
            else:
                current.append(ch)
            i += 1
        parts.append("".join(current).strip())
        return parts

    def _parse_primitive_token(self, token: str) -> Any:
        """Parse a single primitive token (§4 / §B.4)."""
        token = token.strip()
        if not token:
            return ""
        # Quoted string
        if token.startswith('"') and token.endswith('"'):
            return self._unescape(token[1:-1])
        # Boolean / null
        if token == "true":
            return True
        if token == "false":
            return False
        if token == "null":
            return None
        # Number with forbidden leading zeros → string
        if re.match(r'^0\d+$', token):
            return token
        # Integer
        try:
            return int(token)
        except ValueError:
            pass
        # Float
        try:
            return float(token)
        except ValueError:
            pass
        return token

    def _unescape(self, s: str) -> str:
        """Unescape TOON string escapes (§7.1)."""
        return (
            s
            .replace("\\n", "\n")
            .replace("\\r", "\r")
            .replace("\\t", "\t")
            .replace('\\"', '"')
            .replace("\\\\", "\\")
        )


def from_toon(toon_str: str) -> Any:
    """Decode a TOON string back into a Python object."""
    return _ToonDecoder(toon_str.strip()).decode()

def toon_module_list(modules: list[dict]) -> str:
    """
    Serialize the modules list from the extraction JSON into TOON.
    Keeps only fields the planner actually needs to reduce token spend.

    Output format (expanded list with objects as list items):

        [2]:
          - mod: UserManagement
            desc: Handles user listing and role assignment
            reqs[3]: Paginated user list,Role dropdown,Search bar
            ui[4]: Table,Dropdown,Input,Button
            apis[2]: /api/users,/api/roles
          - mod: Dashboard
            ...
    """
    slim: list[dict] = []
    for m in modules:
        entry: dict = {}
        if m.get("module"):
            entry["mod"] = m["module"]
        if m.get("description"):
            entry["desc"] = m["description"]
        reqs = m.get("requirements", [])
        if reqs:
            entry["reqs"] = reqs
        ui = m.get("ui_elements", [])
        if ui:
            entry["ui"] = ui
        apis = m.get("api_endpoints", [])
        if apis:
            entry["apis"] = apis
        slim.append(entry)
    return to_toon(slim)


def toon_rich_extraction(extraction: dict) -> str:
    """
    Convert the full extractor output (modules + shared) into a compact TOON
    representation for the Planner LLM.

    Preserves the full screen → component → field/column/action hierarchy so the
    planner has enough context to produce component-level plans and precise
    similarity queries.

    Extraction shape (from user_story_parser):
      {
        "modules": [
          {
            "name": str, "description": str,
            "screens": [
              {
                "name": str, "screen_type": str, "opens_as": str,
                "purpose": str, "linked_forms": [str],
                "components": [
                  {
                    "type": "toolbar|grid|form|filter_panel|tabs|info_panel",
                    "title": str,
                    "actions":     [{"label": str, ...}],        # toolbar
                    "columns":     [{"name": str, ...}],          # grid
                    "field_groups":[{"fields": [{...}]}],         # form
                    "filters":     [{"name": str, ...}],          # filter_panel
                    "children":    [{"name": str, ...}],          # tabs
                    "fields":      [str],                          # info_panel
                    "validations": [str],                          # form
                  }
                ]
              }
            ]
          }
        ],
        "shared": {
          "enums": [{"name": str, "values": [str]}],
          "global_business_rules": [str],
          "workflows": [{"name": str, "steps": [str], "screens_involved": [str]}]
        }
      }
    """
    modules_raw: list[dict] = extraction.get("modules", [])
    shared: dict = extraction.get("shared", {})

    slim_modules: list[dict] = []
    for mod in modules_raw:
        slim_screens: list[dict] = []

        for screen in mod.get("screens", []):
            slim_comps: list[dict] = []

            for comp in screen.get("components", []):
                ctype = comp.get("type", "")
                slim: dict = {"type": ctype}

                if comp.get("title"):
                    slim["title"] = comp["title"]

                if ctype == "toolbar":
                    labels = [
                        a.get("label", "") for a in comp.get("actions", [])
                        if isinstance(a, dict) and a.get("label")
                    ]
                    if labels:
                        slim["actions"] = labels

                elif ctype == "grid":
                    cols = [
                        c.get("name", c.get("label", ""))
                        for c in comp.get("columns", [])
                        if isinstance(c, dict) and (c.get("name") or c.get("label"))
                    ]
                    if cols:
                        slim["cols"] = cols

                elif ctype == "form":
                    field_names: list[str] = []
                    for fg in comp.get("field_groups", []):
                        if not isinstance(fg, dict):
                            continue
                        for f in fg.get("fields", []):
                            n = (
                                f.get("name", f.get("label", ""))
                                if isinstance(f, dict)
                                else str(f)
                            )
                            if n:
                                field_names.append(n)
                    if field_names:
                        slim["fields"] = field_names
                    valids = comp.get("validations", [])
                    if valids:
                        slim["validations"] = valids

                elif ctype == "filter_panel":
                    filter_names = [
                        f.get("name", f.get("label", f.get("id", "")))
                        for f in comp.get("filters", [])
                        if isinstance(f, dict)
                    ]
                    filter_names = [n for n in filter_names if n]
                    if filter_names:
                        slim["filters"] = filter_names

                elif ctype == "tabs":
                    tab_names: list[str] = []
                    for child in comp.get("children", []):
                        if isinstance(child, dict):
                            n = child.get("name", child.get("title", child.get("tab_name", "")))
                            if n:
                                tab_names.append(n)
                    if tab_names:
                        slim["tabs"] = tab_names

                elif ctype == "info_panel":
                    raw_fields = comp.get("fields", [])
                    flat = [
                        f if isinstance(f, str) else f.get("name", f.get("label", ""))
                        for f in raw_fields
                    ]
                    flat = [f for f in flat if f]
                    if flat:
                        slim["fields"] = flat

                slim_comps.append(slim)

            slim_screen: dict = {
                "name": screen.get("name", ""),
                "type": screen.get("screen_type", "dashboard"),
                "opens_as": screen.get("opens_as", "full-page"),
                "purpose": screen.get("purpose", ""),
            }
            linked = screen.get("linked_forms", [])
            if linked:
                slim_screen["linked_forms"] = linked
            if slim_comps:
                slim_screen["comps"] = slim_comps

            slim_screens.append(slim_screen)

        slim_mod: dict = {
            "name": mod.get("name", ""),
            "desc": mod.get("description", ""),
            "screens": slim_screens,
        }
        slim_modules.append(slim_mod)

    # ── Shared section ────────────────────────────────────────────────────────
    slim_shared: dict = {}

    enums = shared.get("enums", [])
    if enums:
        slim_shared["enums"] = [
            e.get("name", "") for e in enums if isinstance(e, dict) and e.get("name")
        ]

    rules = shared.get("global_business_rules", [])
    if rules:
        # Keep top 8 to stay token-efficient — most critical rules come first
        slim_shared["rules"] = [r for r in rules[:8] if r]

    workflows = shared.get("workflows", [])
    if workflows:
        slim_wf: list[dict] = []
        for wf in workflows:
            if not isinstance(wf, dict):
                continue
            entry: dict = {"name": wf.get("name", "")}
            steps = wf.get("steps", [])
            if steps:
                entry["steps"] = steps
            slim_wf.append(entry)
        if slim_wf:
            slim_shared["workflows"] = slim_wf

    # ── Compose & encode ──────────────────────────────────────────────────────
    data: dict = {}
    if slim_modules:
        data["modules"] = slim_modules
    if slim_shared:
        data["shared"] = slim_shared

    return to_toon(data)

sqlite_store.py
"""
SQLite storage layer for persisting user-story extraction results.

Table: extracted_requirements
  - id              TEXT PRIMARY KEY  (UUID4)
  - user_story_id   TEXT NOT NULL     (identifies the user story / filename)
  - extracted_requirements  TEXT NOT NULL  (JSON blob)
  - created_at      TEXT NOT NULL     (ISO-8601 timestamp)
"""

from __future__ import annotations

import json
import sqlite3
import uuid
from contextlib import contextmanager
from datetime import datetime, timezone
from pathlib import Path
from typing import Any, Generator

DB_PATH = Path(__file__).resolve().parent / "extracted_requirements.db"

_CREATE_TABLE = """
CREATE TABLE IF NOT EXISTS extracted_requirements (
    id                      TEXT PRIMARY KEY,
    user_story_id           TEXT NOT NULL,
    extracted_requirements  TEXT NOT NULL,
    created_at              TEXT NOT NULL
);
"""


@contextmanager
def _get_connection() -> Generator[sqlite3.Connection, None, None]:
    conn = sqlite3.connect(str(DB_PATH))
    conn.row_factory = sqlite3.Row
    try:
        yield conn
        conn.commit()
    except Exception:
        conn.rollback()
        raise
    finally:
        conn.close()


def init_db() -> None:
    """Create the table if it doesn't already exist."""
    with _get_connection() as conn:
        conn.execute(_CREATE_TABLE)


def save_extraction(
    user_story_id: str,
    extracted_requirements: dict | list,
) -> str:
    """
    Persist an extraction result and return the generated UUID.

    Args:
        user_story_id: Identifier for the user story (e.g. filename).
        extracted_requirements: The JSON-serialisable extraction output.

    Returns:
        The UUID (str) of the newly inserted row.
    """
    row_id = str(uuid.uuid4())
    now = datetime.now(timezone.utc).isoformat()
    payload = json.dumps(extracted_requirements, ensure_ascii=False)

    with _get_connection() as conn:
        conn.execute(
            "INSERT INTO extracted_requirements (id, user_story_id, extracted_requirements, created_at) "
            "VALUES (?, ?, ?, ?)",
            (row_id, user_story_id, payload, now),
        )
    return row_id


def get_extraction_by_id(row_id: str) -> dict[str, Any] | None:
    """Fetch a single row by its UUID. Returns None if not found."""
    with _get_connection() as conn:
        cur = conn.execute(
            "SELECT * FROM extracted_requirements WHERE id = ?", (row_id,)
        )
        row = cur.fetchone()
    if row is None:
        return None
    return _row_to_dict(row)


def get_extractions_by_story(user_story_id: str) -> list[dict[str, Any]]:
    """Fetch all rows for a given user_story_id, newest first."""
    with _get_connection() as conn:
        cur = conn.execute(
            "SELECT * FROM extracted_requirements WHERE user_story_id = ? ORDER BY created_at DESC",
            (user_story_id,),
        )
        rows = cur.fetchall()
    return [_row_to_dict(r) for r in rows]


def get_latest_extraction(user_story_id: str) -> dict[str, Any] | None:
    """Fetch the most recent extraction for a given user story."""
    with _get_connection() as conn:
        cur = conn.execute(
            "SELECT * FROM extracted_requirements WHERE user_story_id = ? ORDER BY created_at DESC LIMIT 1",
            (user_story_id,),
        )
        row = cur.fetchone()
    if row is None:
        return None
    return _row_to_dict(row)


def _row_to_dict(row: sqlite3.Row) -> dict[str, Any]:
    d = dict(row)
    d["extracted_requirements"] = json.loads(d["extracted_requirements"])
    return d
